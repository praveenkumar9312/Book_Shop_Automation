    public Inventorylevel(List<Object[]> objectsList) {
        initComponents();
        TableModel model=jTable.getModel();
        int i=0;
        for (Object[] objects : objectsList) {
            model.setValueAt(((Book)objects[0]).getTitle(), i, 0);
            model.setValueAt(objects[2], i, 1);
            model.setValueAt(objects[1], i, 2);
            Book x=(Book) objects[0];
            float proF=(x.getSellprice()-x.getBuyingprice())*(Long)objects[1];
            model.setValueAt(proF, i, 3);
            i++;
        }
    }

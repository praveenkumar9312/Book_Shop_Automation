    private void confirmActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_confirmActionPerformed
        // TODO add your handling code here:
        if(!constraint.checkemail(textmail.getText()))
            JOptionPane.showMessageDialog(textphone, "INVALID EMAIL.");
        else if(!constraint.checkmobile(textphone.getText()))
            JOptionPane.showMessageDialog(textphone, "INVALID PHONE NUMBER.");
        else
        {
            currentemployee.setMail(textmail.getText());
            currentemployee.setPhone(textphone.getText());
            Manager.commit(currentemployee); 
        
            textmail.setVisible(false);
            textphone.setVisible(false);
            l1.setVisible(false);
            l2.setVisible(false);
            confirm.setVisible(false);
            
            mail.setText(currentemployee.getMail());
            phone.setText(currentemployee.getPhone());
            update.setEnabled(true);

        }
    }//GEN-LAST:event_confirmActionPerformed

    public ReservedBooks(Reserves r,clerkpanel c) {
        initComponents();
        title.setText(r.getBook().getTitle());
        reserves=r;
        Authors=Bookf.getAuthors(r.getBook());
        cp=c;
        int i=0;
        Author="";
        for(Author auth : Authors)
        {
            if(" ".equals(auth.getName())||"".equals(auth.getName()))
                continue;
            if(i==0)
            {
                Author = auth.getName();
            }
                
            else 
            {
                Author = Author + " , " + auth.getName();
            }
//            System.out.println(au);
            i++;
        }
        
        author.setText(Author);
        customername.setText(r.getCustomer().getName());
    }

    private void addorderActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_addorderActionPerformed
        
        if((Integer)copies.getValue()==0)
        {JOptionPane.showMessageDialog(addorder, "NUMER OF COPIES CAN NOT BE ZERO.");return;}
        
        Invoice invoice=new Invoice();
        invoice.setBook(currentBook);
        invoice.setCopies((Integer) copies.getValue());
        if(currentBook.getPublisher()!=null&&!Publisherf.isComplete(currentBook.getPublisher()))
        {
            GivePublisherDetails publisherDetails=new GivePublisherDetails(new JFrame(), true,currentBook);
            publisherDetails.setVisible(true);
            publisherDetails.setLocationByPlatform(true);
        }
        if(Publisherf.isComplete((currentBook.getPublisher())))
        {
            Orders os=Ordersf.genrateOrder(currentBook.getPublisher());
            Ordersf.addInvoice(invoice, currentBook.getPublisher());
            addorder.setEnabled(false);

            cancell.setEnabled(false);
            copies.setEnabled(false);    
        }
        
    }//GEN-LAST:event_addorderActionPerformed

    private void cancellActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_cancellActionPerformed
       Library.deleteBookRequests(currentBook);
       addorder.setEnabled(false);
       cancell.setEnabled(false);
       copies.setEnabled(false);
    }//GEN-LAST:event_cancellActionPerformed

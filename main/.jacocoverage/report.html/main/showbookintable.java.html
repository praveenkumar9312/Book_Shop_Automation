    public showbookintable(Book b) {
        
        initComponents();
        
        b_details = b;
        Title=b.getTitle();
        Authors=Bookf.getAuthors(b);
        Price=b.getSellprice();
        price.setText("Rs.  "+String.valueOf(Price));
        title.setText(Title);
        views.setText(String.valueOf(b.getViews()));
        int i=0;
        Author="";
        for(Author auth : Authors)
        {
            if(" ".equals(auth.getName())||"".equals(auth.getName()))
                continue;
            if(i==0)
            {
                Author = auth.getName();
            }
                
            else 
            {
                Author = Author + " , " + auth.getName();
            }
//            System.out.println(au);
            i++;
        }
        
        author.setText(Author);
    }

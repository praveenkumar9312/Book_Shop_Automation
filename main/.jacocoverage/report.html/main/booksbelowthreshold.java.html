    public booksbelowthreshold(Book b) {
        initComponents();
        title.setText(b.getTitle());
        Authors=Bookf.getAuthors(b);
        int i=0;
        Author="";
        for(Author auth : Authors)
        {
            if(" ".equals(auth.getName())||"".equals(auth.getName()))
                continue;
            if(i==0)
            {
                Author = auth.getName();
            }
                
            else 
            {
                Author = Author + " , " + auth.getName();
            }
//            System.out.println(au);
            i++;
        }
        
        author.setText(Author);     
        threshold.setText(String.valueOf(thres-b.getCopiesPresent()));
    }

    private void signupActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_signupActionPerformed
        // TODO add your handling code here:
        System.out.println("name="+name.getText());
        if("".equals(mail.getText()) || "".equals(name.getText()) || "".equals(phone.getText()) || "".equals(String.copyValueOf(signuppass.getPassword())) || "".equals(String.copyValueOf(confirmpass.getPassword())))
            JOptionPane.showMessageDialog(null, "YOU CAN NOT LEAVE ANY FIELD EMPTY.");
        else if(!constraint.checkemail(mail.getText()))
        {
            JOptionPane.showMessageDialog(null, "INVALID EMAIL ADDRESS.\nPLEASE TRY AGAIN");
            mail.setText(null);
         }
        else if(!constraint.checkmobile(phone.getText()))
        {
            JOptionPane.showMessageDialog(null, "INVALID PHONE NUMBER.\nPLEASE TRY AGAIN");
            phone.setText(null);
        }
        else if(String.valueOf(signuppass.getPassword() )== null ? String.valueOf(confirmpass.getPassword()) == null :String.valueOf( signuppass.getPassword()).equals(String.valueOf(confirmpass.getPassword())))
        {
            Employee e = new Employee();
            e.setLoginid(name.getText());
            e.setName(name.getText());
            e.setMail(mail.getText());
            e.setPhone(phone.getText());
            e.setPassword(String.copyValueOf(signuppass.getPassword()));
            
            Employee temp= new Employee();
            
            if(type.getSelectedIndex()==0)
                temp = Shopfeatures.registerEmployee(e);
            else if(type.getSelectedIndex()==1)
                temp = Shopfeatures.registerClerk(e);
            else if(type.getSelectedIndex()==2)
                temp = Shopfeatures.registerManager(e);
            else if(type.getSelectedIndex()==3)
                temp = Shopfeatures.registerOwner(e);
            
            if(temp!=null)
            {
                JOptionPane.showMessageDialog(null, "REGISTERED SUCCESSFULLY.");
                name.setText(null);
                mail.setText(null);
                phone.setText(null);
                signuppass.setText(null);
                confirmpass.setText(null);
            }
            else
            {
                name.setText(null);
            }
                
        }
        else
        {
            JOptionPane.showMessageDialog(null, "CONFIRM PASSWORD SHOULD BE SAME AS PASSWORD.");
            signuppass.setText(null);
            confirmpass.setText(null);
        }
        
    }//GEN-LAST:event_signupActionPerformed

    private void loginActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_loginActionPerformed
        // TODO add your handling code here:
        if("".equals(loginid.getText()) || "".equals(String.valueOf(loginpass.getPassword())))
            JOptionPane.showMessageDialog(null, "YOU CAN NOT LEAVE ANY FIELD EMPTY.");
        else if(Shopfeatures.getEmployeefromLogin(loginid.getText())!=null)
        {
            Employee e =Shopfeatures.getEmployeefromLogin(loginid.getText());
            if(e.getPassword() == null ? String.valueOf(loginpass.getPassword()) == null : e.getPassword().equals(String.valueOf(loginpass.getPassword())))
            {
//                main.dispose();
//                shop a = new shop();
//                a.setVisible(true);
                  employeedetails ed = new employeedetails(e);
                  main.setscrollpanecategory(ed);
                  if(e.getType()==Employeef.CLERK_TYPE)
                  {
                      main.addclerkpanel();
                  }  
                  else if(e.getType()==Employeef.EMPLOYEE_TYPE)
                  {
                        main.addincompletebooks();
                  }
                  else if(e.getType()==Employeef.MANAGER_TYPE)
                  {
                      main.addmanagerpanel();
                  }
                  else if(e.getType()==Employeef.OWNER_TYPE)
                  {
                      main.addownerpanel();
                  }
                  UserSession.storeLoggedInEmployee(e);
                  
            }else
                JOptionPane.showMessageDialog(null, "INCORRECT PASSWORD.\nPLEASE TRY AGAIN.");
        }
        else
             JOptionPane.showMessageDialog(null, "YOU ARE NOT REGISTERED YET.\nPLEASE REGISTER FIRST.");
    }//GEN-LAST:event_loginActionPerformed

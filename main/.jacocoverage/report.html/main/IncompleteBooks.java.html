    public IncompleteBooks(Book bobj) {
        initComponents();
        b=bobj;
        title.setText(bobj.getTitle());
        Authors = Bookf.getAuthors(b);
         int i=0;
        for(Author auth : Authors)
        {
            if("".equals(auth.getName()))
                continue;
            if(i==0)
                Author = auth.getName();
            else 
                Author = Author + " , " + auth.getName();
            i++;
        }
        
        author.setText(Author);
    }

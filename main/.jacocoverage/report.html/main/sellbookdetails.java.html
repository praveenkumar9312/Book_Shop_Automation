        authordetail.setText(Bookf.getAuthors(book1).get(0).getName());
        rack.setText("On rack "+extrafunctions.getracknumber(book1));
    }

    private void confirmActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_confirmActionPerformed
        // TODO add your handling code here:
        if(currentBook.getCopiesPresent() < Integer.parseInt(textcopies.getText()))
            JOptionPane.showMessageDialog(textcopies, "BOOKSHOP DO NOT HAVE THIS MUCH COPIES FOR THE BOOK.");
        else if( textcopies.getText()==null)
            JOptionPane.showMessageDialog(textcopies, "PLEASE ENTER NUMBER OF COPIES.");
        else
        {
            c.addtablerow(currentBook,Integer.parseInt(textcopies.getText()));
            this.dispose();
        }
        count++;
    }//GEN-LAST:event_confirmActionPerformed

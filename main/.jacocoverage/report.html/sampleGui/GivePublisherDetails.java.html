    private void saveButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_saveButtonActionPerformed
        System.out.println("I'm here           ");
        System.out.println(publisherField.getEditor().getItem().toString());
        publisher=new Publisher();
        if(!"".equals(vendorField.getText())||!"".equals(publisherField.getEditor().getItem().toString())||!"".equals(addressArea.getText()))
        {
            
            publisher.setName(publisherField.getEditor().getItem().toString());
            
            publisher.setVendor(vendorField.getText());
            publisher.setMail(mailField.getText());
            publisher.setAddress(addressArea.getText());
            publisher.setPhone(phoneField.getText());
            System.out.println("sending to function "+publisher);
            publisher=Publisherf.addPublisher(publisher);
            currentBook.setPublisher(publisher);
            Manager.commit(currentBook);
            JOptionPane.showMessageDialog(null, "Succesfully updated Publisher Details");
            this.dispose();
        }
        else
            JOptionPane.showMessageDialog(saveButton, "Please give correct information");
        
        
    }//GEN-LAST:event_saveButtonActionPerformed
